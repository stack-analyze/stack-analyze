const timeout = 1e3;

 * @property {string} Youtube.recomendation
 * 
 * @typedef {Object} Twitch
 * @property {string} Twitch.user

 * 
 * @typedef {Object} Project
 * @property {string} Project.name
 * @property {string} Project.desc
 */
